## 1. Complete `signup` in **`actions/auth-actions.js`**

```js
'use server';

import bcrypt from 'bcrypt';
import { db } from '../lib/database'; // adjust import to your DB helper

export async function signup(prevState, formData) {
  // 1) Extract
  const email    = formData.get('email')?.toString().trim() || '';
  const password = formData.get('password')?.toString().trim() || '';

  // 2) Validate
  const errors = {};
  if (!email.includes('@')) {
    errors.email = 'Please enter a valid email address.';
  }
  if (password.length < 8) {
    errors.password = 'Password must be at least 8 characters long.';
  }
  if (Object.keys(errors).length > 0) {
    return { errors };
  }

  // 3) Hash & insert
  const hash = await bcrypt.hash(password, 12);
  await db.prepare(
    'INSERT INTO users (email, password) VALUES (?, ?)'
  ).run(email, hash);

  // 4) Indicate success (no errors)
  return { success: true };
}
```

> **Notes:**
> 
> - We kept the twoâ€‘argument signature (`prevState, formData`) so `useFormState` can wire up errors.
>     
> - You may want to catch uniqueâ€‘email constraint violations and return an error object in that case, too.
